def self.from_string(string,blank_is_nil:true)
  if string.nil?
      return nil
  end
  if string.to_s.strip == "" && blank_is_nil
    return nil
  end
  self.new(string)
end

def +(other)
  if other.kind_of?(RichString)
    RichString.new(self.to_s + other.to_s)
  elsif other.kind_of?(String)
    self.to_s + other
  else
    super(other)
  end
end

def ==(other)
  if other.kind_of?(RichString)
    self.to_s == other.to_s
  elsif other.kind_of?(String)
    self.to_s == other
  else
    false
  end
end
