def exit_status(&when_error_message)
  when_error_message ||= ->(*){}
  case @actual_result
  in Integer
    @actual_result
  in Brut::CLI::SystemExecError
    when_error_message.(@actual_result.message)
    @actual_result.exit_status
  in Brut::CLI::Error
    when_error_message.(@actual_result.message)
    1
  else
    0
  end
end
