def create!
  if @project_root.exist?
    raise "Project root #{@project_root} already exists"
  end
  operations = [ Ops::Mkdir.new(@project_root) ]           + 
                 copy_files(@templates_dir, @project_root)

  operations.each do |operation|
    operation.call
  end
end
