def call
  if dry_run?
    puts "Render '#{@destination_file}'"
    return
  end
  template = File.read(@erb)
  File.open(@destination_file, "w") do |file|
    file.puts ERB.new(
      template,
      trim_mode: "-"
    ).result(
      @erb_binding.instance_eval { binding }
    )
  end
end
