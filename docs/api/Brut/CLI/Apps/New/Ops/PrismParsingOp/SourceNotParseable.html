def initialize(tree_errors:, file:)
  error_message = tree_errors.map { |error|
    "#{error.message} (line #{error.location.start_line}, column #{error.location.start_column})"
  }.join(", ")
  super("Failed to parse file '#{file}': #{error_message}")
end
