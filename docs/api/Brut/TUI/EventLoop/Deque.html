def pop(timeout:)
  @mutex.synchronize {
    deadline = Time.now + timeout
    while @array.empty?
      remaining = deadline - Time.now
      if remaining <= 0
        return nil 
      end
      @condition_variable.wait(@mutex, remaining)
    end
    @array.shift
  }
end
