def setup!
  Faker::Config.locale = :en
  FactoryBot.definition_file_paths = [
    Brut.container.app_specs_dir / "factories"
  ]
  FactoryBot.define do
    to_create { |instance| instance.save }
  end
  FactoryBot.find_definitions
end
